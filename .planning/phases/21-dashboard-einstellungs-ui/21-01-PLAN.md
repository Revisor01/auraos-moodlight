    ```javascript
    // ===== EINSTELLUNGEN (UI-01/02/03/04/05) =====
    var apiKeyEditing = false;

    async function loadSettings() {
      try {
        var res  = await fetch('/api/moodlight/settings');
        var data = await res.json();
        if (data.status !== 'success') return;

        var s = data.settings;

        // Analyse-Felder befüllen
        document.getElementById('settingInterval').value  = s.analysis_interval_minutes;
        document.getElementById('settingHeadlines').value = s.headlines_per_source;

        // API Key: maskierten Wert als Platzhalter setzen, Feld gesperrt
        var keyEl = document.getElementById('settingApiKey');
        keyEl.value    = s.anthropic_api_key || '';
        keyEl.readOnly = true;
        keyEl.type     = 'password';
        apiKeyEditing  = false;
        document.getElementById('editKeyBtn').textContent = 'Bearbeiten';

        settingsLoaded = true;
      } catch (e) {
        console.error('Fehler beim Laden der Einstellungen:', e);
      }
    }

    async function saveAnalysis() {
      var btn = document.getElementById('saveAnalysisBtn');
      var msg = document.getElementById('analysisMsg');
      showSettingsMsg(msg, '', '');

      var minutes   = parseFloat(document.getElementById('settingInterval').value);
      var headlines = parseInt(document.getElementById('settingHeadlines').value, 10);

      if (isNaN(minutes) || minutes < 1) {
        showSettingsMsg(msg, 'error', 'Analyse-Frequenz muss mindestens 1 Minute betragen.');
        return;
      }
      if (isNaN(headlines) || headlines < 1) {
        showSettingsMsg(msg, 'error', 'Headlines pro Quelle muss mindestens 1 sein.');
        return;
      }

      btn.disabled = true;
      try {
        var res  = await fetch('/api/moodlight/settings', {
          method: 'PUT',
          headers: {'Content-Type': 'application/json'},
          body: JSON.stringify({
            analysis_interval_minutes: minutes,
            headlines_per_source: headlines
          })
        });
        var data = await res.json();
        if (res.ok && data.status === 'success') {
          showSettingsMsg(msg, 'success', 'Gespeichert: ' + (data.message || 'Einstellungen aktualisiert.'));
        } else {
          showSettingsMsg(msg, 'error', data.message || 'Fehler beim Speichern.');
        }
      } catch (e) {
        showSettingsMsg(msg, 'error', 'Netzwerkfehler — API nicht erreichbar.');
        console.error(e);
      } finally {
        btn.disabled = false;
      }
    }

    function toggleApiKeyEdit() {
      var keyEl  = document.getElementById('settingApiKey');
      var btnEl  = document.getElementById('editKeyBtn');
      if (!apiKeyEditing) {
        // Editier-Modus aktivieren: Feld leeren, Klartext-Eingabe
        keyEl.value    = '';
        keyEl.readOnly = false;
        keyEl.type     = 'text';
        keyEl.focus();
        btnEl.textContent = 'Abbrechen';
        apiKeyEditing  = true;
      } else {
        // Abbrechen: Feld wieder sperren, maskierten Wert zurücksetzen
        settingsLoaded = false;
        loadSettings();
        btnEl.textContent = 'Bearbeiten';
        apiKeyEditing  = false;
      }
    }

    async function saveAuth() {
      var btn   = document.getElementById('saveAuthBtn');
      var msg   = document.getElementById('authMsg');
      showSettingsMsg(msg, '', '');

      var newKey = document.getElementById('settingApiKey').value.trim();

      // Wenn nicht im Editier-Modus und kein neuer Key: nichts tun
      if (!apiKeyEditing && !newKey) {
        showSettingsMsg(msg, 'error', 'Aktiviere zuerst "Bearbeiten" um den API Key zu ändern.');
        return;
      }

      btn.disabled = true;
      try {
        var body = {};
        if (apiKeyEditing && newKey) {
          body.anthropic_api_key = newKey;
        }

        if (Object.keys(body).length === 0) {
          showSettingsMsg(msg, 'error', 'Kein neuer API Key eingegeben.');
          btn.disabled = false;
          return;
        }

        var res  = await fetch('/api/moodlight/settings', {
          method: 'PUT',
          headers: {'Content-Type': 'application/json'},
          body: JSON.stringify(body)
        });
        var data = await res.json();

        if (res.ok && data.status === 'success') {
          showSettingsMsg(msg, 'success', 'API Key gespeichert.');
          // Editier-Modus beenden, neu laden
          apiKeyEditing  = false;
          settingsLoaded = false;
          loadSettings();
        } else {
          showSettingsMsg(msg, 'error', data.message || 'Fehler beim Speichern.');
        }
      } catch (e) {
        showSettingsMsg(msg, 'error', 'Netzwerkfehler — API nicht erreichbar.');
        console.error(e);
      } finally {
        btn.disabled = false;
      }
    }

    async function savePassword() {
      var btn  = document.getElementById('savePasswordBtn');
      var msg  = document.getElementById('passwordMsg');
      showSettingsMsg(msg, '', '');

      var current = document.getElementById('currentPassword').value;
      var newPw   = document.getElementById('newPassword').value;
      var confirm = document.getElementById('confirmPassword').value;

      if (!current) {
        showSettingsMsg(msg, 'error', 'Aktuelles Passwort eingeben.');
        return;
      }
      if (!newPw) {
        showSettingsMsg(msg, 'error', 'Neues Passwort eingeben.');
        return;
      }
      if (newPw !== confirm) {
        showSettingsMsg(msg, 'error', 'Passwörter stimmen nicht überein.');
        return;
      }
      if (newPw.length < 6) {
        showSettingsMsg(msg, 'error', 'Neues Passwort muss mindestens 6 Zeichen lang sein.');
        return;
      }

      btn.disabled = true;
      try {
        var res  = await fetch('/api/moodlight/settings', {
          method: 'PUT',
          headers: {'Content-Type': 'application/json'},
          body: JSON.stringify({
            admin_password:   newPw,
            current_password: current
          })
        });
        var data = await res.json();

        if (res.ok && data.status === 'success') {
          showSettingsMsg(msg, 'success', 'Passwort erfolgreich geändert.');
          document.getElementById('currentPassword').value = '';
          document.getElementById('newPassword').value     = '';
          document.getElementById('confirmPassword').value = '';
        } else {
          var errorText = data.message || 'Fehler beim Ändern.';
          // 403 = falsches aktuelles Passwort
          if (res.status === 403) {
            errorText = 'Aktuelles Passwort ist falsch.';
          }
          showSettingsMsg(msg, 'error', errorText);
        }
      } catch (e) {
        showSettingsMsg(msg, 'error', 'Netzwerkfehler — API nicht erreichbar.');
        console.error(e);
      } finally {
        btn.disabled = false;
      }
    }

    function showSettingsMsg(el, type, text) {
      el.className = 'settings-msg' + (type ? ' ' + type : '');
      el.textContent = text;
      el.style.display = text ? 'block' : 'none';
    }
    ```
