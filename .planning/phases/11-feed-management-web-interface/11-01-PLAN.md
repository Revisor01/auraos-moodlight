JavaScript (inline am Ende von body):
```javascript
const API = '/api/moodlight/feeds';

// Hilfsfunktionen
function formatDate(iso) {
  if (!iso) return '—';
  const d = new Date(iso);
  return d.toLocaleString('de-DE', {
    day: '2-digit', month: '2-digit', year: 'numeric',
    hour: '2-digit', minute: '2-digit'
  });
}

function setLoading(state) {
  document.getElementById('loadingMsg').hidden = !state;
  document.getElementById('tableWrapper').style.opacity = state ? '0.5' : '1';
}

// Feeds laden und Tabelle befuellen
async function loadFeeds() {
  setLoading(true);
  try {
    const res = await fetch(API);
    const data = await res.json();
    const feeds = data.feeds || [];

    document.getElementById('feedCount').textContent = feeds.length;
    document.getElementById('emptyMsg').hidden = feeds.length > 0;

    const tbody = document.getElementById('feedBody');
    tbody.innerHTML = '';
    feeds.forEach(feed => {
      const tr = document.createElement('tr');
      tr.dataset.id = feed.id;
      tr.innerHTML = `
        <td>${escapeHtml(feed.name)}</td>
        <td><a href="${escapeHtml(feed.url)}" target="_blank" rel="noopener">${escapeHtml(truncateUrl(feed.url))}</a></td>
        <td>${formatDate(feed.last_fetched_at)}</td>
        <td class="${feed.error_count > 0 ? 'error-count' : ''}">${feed.error_count}</td>
        <td><button class="btn-delete" data-id="${feed.id}" data-name="${escapeHtml(feed.name)}">Loeschen</button></td>
      `;
      tbody.appendChild(tr);
    });

    // Delete-Handler einhaengen
    tbody.querySelectorAll('.btn-delete').forEach(btn => {
      btn.addEventListener('click', () => deleteFeed(btn.dataset.id, btn.dataset.name));
    });

  } catch (e) {
    console.error('Fehler beim Laden der Feeds:', e);
  } finally {
    setLoading(false);
    document.getElementById('loadingMsg').hidden = true;
  }
}

// Feed loeschen (kein confirm-Dialog)
async function deleteFeed(id, name) {
  const btn = document.querySelector(`.btn-delete[data-id="${id}"]`);
  if (btn) { btn.disabled = true; btn.textContent = '...'; }

  try {
    const res = await fetch(`${API}/${id}`, { method: 'DELETE' });
    if (res.status === 204) {
      // Zeile sofort aus DOM entfernen
      const row = document.querySelector(`tr[data-id="${id}"]`);
      if (row) row.remove();
      const count = document.getElementById('feedCount');
      count.textContent = parseInt(count.textContent || '0') - 1;
      const tbody = document.getElementById('feedBody');
      document.getElementById('emptyMsg').hidden = tbody.children.length > 0;
    } else {
      const data = await res.json();
      alert(`Fehler: ${data.message || 'Unbekannter Fehler'}`);
      if (btn) { btn.disabled = false; btn.textContent = 'Loeschen'; }
    }
  } catch (e) {
    console.error('Fehler beim Loeschen:', e);
    if (btn) { btn.disabled = false; btn.textContent = 'Loeschen'; }
  }
}

// Feed hinzufuegen
document.getElementById('addForm').addEventListener('submit', async (e) => {
  e.preventDefault();
  const errorEl = document.getElementById('addError');
  const successEl = document.getElementById('addSuccess');
  const addBtn = document.getElementById('addBtn');

  errorEl.hidden = true;
  successEl.hidden = true;
  addBtn.disabled = true;
  addBtn.textContent = 'Pruefe URL...';

  const name = document.getElementById('feedName').value.trim();
  const url = document.getElementById('feedUrl').value.trim();

  try {
    const res = await fetch(API, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ name, url })
    });
    const data = await res.json();

    if (res.status === 201) {
      successEl.textContent = `Feed "${name}" erfolgreich hinzugefuegt.`;
      successEl.hidden = false;
      document.getElementById('feedName').value = '';
      document.getElementById('feedUrl').value = '';
      await loadFeeds(); // Tabelle aktualisieren
    } else {
      errorEl.textContent = data.message || 'Fehler beim Hinzufuegen.';
      errorEl.hidden = false;
    }
  } catch (e) {
    errorEl.textContent = 'Netzwerkfehler — API nicht erreichbar.';
    errorEl.hidden = false;
  } finally {
    addBtn.disabled = false;
    addBtn.textContent = 'Hinzufuegen';
  }
});

function escapeHtml(str) {
  const d = document.createElement('div');
  d.appendChild(document.createTextNode(str || ''));
  return d.innerHTML;
}

function truncateUrl(url) {
  return url.length > 60 ? url.slice(0, 57) + '...' : url;
}

// Initial laden
loadFeeds();
```
