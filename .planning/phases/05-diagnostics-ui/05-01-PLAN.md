function loadVersion() {
    fetch('/api/firmware-version')
        .then(r => r.text())
        .then(v => {
            const ver = v.trim();
            document.getElementById('version').textContent = 'v' + ver;
        })
        .catch(() => {
            document.getElementById('version').textContent = 'v?.?';
        });
}

**3. JavaScript-Funktionen (UI-01, UI-02) — im `<script>` Block vor der `loadMetrics` Funktion einfuegen:**

```javascript
function onCombinedFileSelected(input) {
    const label = document.getElementById('combined-file-label');
    const info = document.getElementById('combined-file-info');
    const btn = document.getElementById('combined-update-btn');
    if (input.files.length > 0) {
        const file = input.files[0];
        label.innerHTML = '<i class="fas fa-file-archive" aria-hidden="true"></i> ' + file.name;
        info.textContent = 'Groesse: ' + (file.size / 1024).toFixed(1) + ' KB';
        btn.disabled = false;
    } else {
        label.innerHTML = '<i class="fas fa-file-archive" aria-hidden="true"></i> Datei waehlen...';
        info.textContent = '';
        btn.disabled = true;
    }
}

function startCombinedUpdate() {
    const fileInput = document.getElementById('combined-file');
    if (!fileInput.files.length) { alert('Bitte zuerst eine Datei auswaehlen.'); return; }
    if (!confirm('Combined Update starten? Dies aktualisiert UI und Firmware.')) return;

    const file = fileInput.files[0];
    const btn = document.getElementById('combined-update-btn');
    const progress = document.getElementById('combined-progress');
    const stepText = document.getElementById('combined-step-text');
    const bar = document.getElementById('combined-progress-bar');
    const statusText = document.getElementById('combined-status-text');

    btn.disabled = true;
    progress.style.display = 'block';

    // Step 1/2: UI Upload
    stepText.textContent = 'Schritt 1/2: UI-Installation...';
    bar.style.width = '0%';
    bar.textContent = '0%';
    statusText.textContent = 'Lade UI-Dateien hoch...';

    uploadFile('/update/combined-ui', file, bar, function(success, response) {
        if (!success) {
            stepText.textContent = 'Fehler bei Schritt 1/2';
            statusText.textContent = 'UI-Upload fehlgeschlagen: ' + response;
            statusText.style.color = 'var(--negative-color)';
            btn.disabled = false;
            return;
        }
        // Step 2/2: Firmware Flash
        stepText.textContent = 'Schritt 2/2: Firmware-Flash...';
        bar.style.width = '0%';
        bar.textContent = '0%';
        statusText.textContent = 'Flashe Firmware... (Geraet startet danach automatisch neu)';

        uploadFile('/update/combined-firmware', file, bar, function(fwSuccess, fwResponse) {
            if (fwSuccess) {
                stepText.textContent = 'Update abgeschlossen!';
                bar.style.width = '100%';
                bar.textContent = '100%';
                statusText.textContent = 'Firmware geflasht. Geraet startet neu... Seite wird in 15 Sekunden neu geladen.';
                statusText.style.color = 'var(--positive-color)';
                setTimeout(function() { window.location.reload(); }, 15000);
            } else {
                stepText.textContent = 'Fehler bei Schritt 2/2';
                statusText.textContent = 'Firmware-Flash fehlgeschlagen: ' + fwResponse;
                statusText.style.color = 'var(--negative-color)';
                btn.disabled = false;
            }
        });
    });
}

function uploadFile(url, file, barEl, callback) {
    var xhr = new XMLHttpRequest();
    var formData = new FormData();
    formData.append('update', file, file.name);

    xhr.upload.addEventListener('progress', function(e) {
        if (e.lengthComputable) {
            var pct = Math.round((e.loaded / e.total) * 100);
            barEl.style.width = pct + '%';
            barEl.textContent = pct + '%';
        }
    });

    xhr.addEventListener('load', function() {
        if (xhr.status === 200) {
            callback(true, xhr.responseText);
        } else {
            callback(false, xhr.status + ' ' + xhr.responseText);
        }
    });

    xhr.addEventListener('error', function() {
        // Bei Firmware-Flash bricht Connection ab weil ESP neustartet — das ist Erfolg
        if (url.indexOf('firmware') !== -1) {
            callback(true, 'OK');
        } else {
            callback(false, 'Verbindung fehlgeschlagen');
        }
    });

    xhr.addEventListener('timeout', function() {
        if (url.indexOf('firmware') !== -1) {
            callback(true, 'OK');
        } else {
            callback(false, 'Timeout');
        }
    });

    xhr.timeout = 120000;
    xhr.open('POST', url);
    xhr.send(formData);
}

function legacyUpload(url, inputId) {
    var input = document.getElementById(inputId);
    if (!input.files.length) { alert('Bitte zuerst eine Datei auswaehlen.'); return; }
    var formData = new FormData();
    formData.append('update', input.files[0], input.files[0].name);
    fetch(url, { method: 'POST', body: formData })
        .then(function(r) { return r.text(); })
        .then(function(t) { alert('Ergebnis: ' + t); })
        .catch(function(e) { alert('Fehler: ' + e.message); });
}
```
