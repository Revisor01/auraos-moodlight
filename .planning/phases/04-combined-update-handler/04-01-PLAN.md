Neue Header-only Datei `firmware/src/ChunkStream.h` erstellen. Die Klasse erbt von `Stream` (Arduino) und dient als Adapter zwischen dem Push-basierten WebServer-Upload-Callback und der Pull-basierten ESP32-targz-Library.

Architektur: Die ESP32-targz-Library laeuft in einem separaten FreeRTOS-Task (Core 0) und liest synchron von `ChunkStream`. Der WebServer-Task (Core 1) fuettert den Stream via `feed()`. Synchronisation ueber zwei FreeRTOS-Semaphore:
- `_dataReady`: signalisiert dass neue Daten verfuegbar sind (feed -> readBytes)
- `_spaceReady`: signalisiert dass der Buffer wieder Platz hat (readBytes -> feed)

Konkrete Implementierung:

```cpp
#ifndef CHUNK_STREAM_H
#define CHUNK_STREAM_H

#include <Arduino.h>
#include <Stream.h>
#include "freertos/FreeRTOS.h"
#include "freertos/semphr.h"

class ChunkStream : public Stream {
public:
    ChunkStream(size_t bufferSize = 4096) : _bufSize(bufferSize) {
        _buffer = (uint8_t*)malloc(bufferSize);
        _readPos = 0;
        _writePos = 0;
        _available = 0;
        _eof = false;
        _dataReady = xSemaphoreCreateBinary();
        _spaceReady = xSemaphoreCreateBinary();
        xSemaphoreGive(_spaceReady); // Initial: Buffer ist leer = Platz vorhanden
    }

    ~ChunkStream() {
        if (_buffer) free(_buffer);
        if (_dataReady) vSemaphoreDelete(_dataReady);
        if (_spaceReady) vSemaphoreDelete(_spaceReady);
    }

    // --- Aufgerufen vom WebServer-Upload-Task (Core 1) ---

    // Daten in den Ring-Buffer schreiben. Blockiert bis genug Platz ist.
    void feed(const uint8_t* data, size_t len) {
        size_t written = 0;
        while (written < len) {
            // Warten bis Platz im Buffer
            while (_available >= _bufSize) {
                xSemaphoreTake(_spaceReady, pdMS_TO_TICKS(100));
            }
            // Schreibe so viel wie moeglich
            size_t canWrite = min(len - written, _bufSize - _available);
            for (size_t i = 0; i < canWrite; i++) {
                _buffer[_writePos] = data[written + i];
                _writePos = (_writePos + 1) % _bufSize;
            }
            _available += canWrite;
            written += canWrite;
            xSemaphoreGive(_dataReady); // Signalisiere: Daten verfuegbar
        }
    }

    void setEOF() {
        _eof = true;
        xSemaphoreGive(_dataReady); // Aufwecken falls Library auf Daten wartet
    }

    void reset() {
        _readPos = 0;
        _writePos = 0;
        _available = 0;
        _eof = false;
        // Semaphore zuruecksetzen
        xSemaphoreTake(_dataReady, 0);
        xSemaphoreGive(_spaceReady);
    }

    // --- Stream-Interface (aufgerufen von ESP32-targz Library-Task, Core 0) ---

    int available() override {
        return _available > 0 ? _available : (_eof ? 0 : -1);
        // Note: Library prueft available() > 0, bei -1 wartet sie mit vTaskDelay
        // Bei 0 + eof interpretiert sie "fertig"
        // KORREKTUR: available() muss >= 0 sein laut Stream-Interface.
        // Die Library hat eigene Timeout-Logik mit targz_read_timeout (10s).
    }

    int read() override {
        if (_available == 0 && !_eof) {
            // Warten auf Daten (max 10s, passt zu targz_read_timeout)
            xSemaphoreTake(_dataReady, pdMS_TO_TICKS(10000));
        }
        if (_available == 0) return -1;
        uint8_t b = _buffer[_readPos];
        _readPos = (_readPos + 1) % _bufSize;
        _available--;
        if (_available < _bufSize / 2) {
            xSemaphoreGive(_spaceReady); // Signalisiere: Platz frei
        }
        return b;
    }

    size_t readBytes(char* buf, size_t len) override {
        size_t totalRead = 0;
        while (totalRead < len) {
            if (_available == 0) {
                if (_eof) break;
                xSemaphoreTake(_dataReady, pdMS_TO_TICKS(10000));
                if (_available == 0 && _eof) break;
                if (_available == 0) break; // Timeout
            }
            size_t canRead = min(len - totalRead, _available);
            for (size_t i = 0; i < canRead; i++) {
                buf[totalRead + i] = _buffer[_readPos];
                _readPos = (_readPos + 1) % _bufSize;
            }
            _available -= canRead;
            totalRead += canRead;
            if (_available < _bufSize / 2) {
                xSemaphoreGive(_spaceReady);
            }
        }
        return totalRead;
    }

    int peek() override {
        if (_available == 0) return -1;
        return _buffer[_readPos];
    }

    size_t write(uint8_t) override { return 0; } // Nicht benoetigt

    bool isEOF() const { return _eof && _available == 0; }

private:
    uint8_t* _buffer;
    size_t _bufSize;
    volatile size_t _readPos;
    volatile size_t _writePos;
    volatile size_t _available;
    volatile bool _eof;
    SemaphoreHandle_t _dataReady;
    SemaphoreHandle_t _spaceReady;
};

#endif // CHUNK_STREAM_H
```

WICHTIG — `available()` muss korrigiert werden: Das Arduino Stream-Interface erwartet `>= 0` als Rueckgabewert. Wenn keine Daten da sind und kein EOF: `return 0` (nicht -1). Die ESP32-targz-Library hat eine eigene Timeout-Logik (`targz_read_timeout = 10000ms`) mit `vTaskDelay(1)` Polling in `LibUnpacker.cpp:1183-1193`, die bei `available() == 0` wartet.

Korrigierte `available()`:
```cpp
int available() override {
    return (int)_available;
}
```

Die Library unterscheidet nicht zwischen "noch keine Daten" und "fertig" ueber available() — sie hat einen internen Timeout. `readBytes()` gibt 0 zurueck wenn EOF+leer, was die Library als Stream-Ende interpretiert.

Buffer-Groesse: 4096 Bytes als Default. Der ESP32 hat ~320KB Heap, die Library braucht ~40KB fuer Extraktion. 4KB Ring-Buffer ist konservativ sicher. Falls instabil: auf 8KB erhoehen.
