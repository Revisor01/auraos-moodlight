**Schritt 2 — JavaScript einfügen:**

Füge direkt vor dem schließenden `</body>`-Tag folgendes Script-Block ein (nach den bestehenden Chart.js-Initialisierungen, falls vorhanden, oder als letztes Element):

```html
    <script>
    // Headlines-Sektion laden (PAGE-01)
    (function() {
        var API_URL = 'https://analyse.godsapp.de/api/moodlight/headlines?limit=20';

        function scoreToColor(score) {
            if (score <= -0.4) return '#dc3545';
            if (score <  0)   return '#fd7e14';
            if (score === 0)  return '#17a2b8';
            if (score <  0.4) return '#6f42c1';
            return '#8A2BE2';
        }

        function renderHeadlines(headlines) {
            var list = document.getElementById('headlines-list');
            if (!list) return;
            if (!headlines || headlines.length === 0) {
                list.innerHTML = '<li class="headlines-error">Keine Schlagzeilen verfügbar.</li>';
                return;
            }
            var items = '';
            for (var i = 0; i < headlines.length; i++) {
                var h = headlines[i];
                var score = typeof h.sentiment_score === 'number'
                    ? (h.sentiment_score >= 0 ? '+' : '') + h.sentiment_score.toFixed(2)
                    : '--';
                var color = typeof h.sentiment_score === 'number' ? scoreToColor(h.sentiment_score) : '#6c757d';
                var feed = h.feed_name || h.source_name || '';
                items += '<li class="headline-item">';
                items += '<span class="headline-score" style="color:' + color + ';">' + score + '</span>';
                items += '<span class="headline-text">' + h.headline + '</span>';
                if (feed) {
                    items += '<span class="headline-feed">' + feed + '</span>';
                }
                items += '</li>';
            }
            list.innerHTML = items;
        }

        function loadHeadlines() {
            fetch(API_URL)
                .then(function(r) {
                    if (!r.ok) throw new Error('HTTP ' + r.status);
                    return r.json();
                })
                .then(function(data) {
                    if (data.status === 'success') {
                        renderHeadlines(data.headlines);
                    } else {
                        var list = document.getElementById('headlines-list');
                        if (list) list.innerHTML = '<li class="headlines-error">Schlagzeilen konnten nicht geladen werden.</li>';
                    }
                })
                .catch(function() {
                    var list = document.getElementById('headlines-list');
                    if (list) list.innerHTML = '<li class="headlines-error">Backend nicht erreichbar — Schlagzeilen nicht verfügbar.</li>';
                });
        }

        if (document.readyState === 'loading') {
            document.addEventListener('DOMContentLoaded', loadHeadlines);
        } else {
            loadHeadlines();
        }
    })();
    </script>
```
