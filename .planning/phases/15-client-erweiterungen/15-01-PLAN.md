```html
<script>
(function() {
    var HEADLINES_URL = 'https://analyse.godsapp.de/api/moodlight/headlines?limit=20';

    function scoreColor(score) {
        if (score <= -0.4) return '#e53e3e';
        if (score < 0)    return '#dd6b20';
        if (score === 0)  return '#3182ce';
        if (score < 0.4)  return '#6b46c1';
        return '#805ad5';
    }

    function renderHeadlines(headlines) {
        var container = document.getElementById('headlines-list');
        if (!headlines || headlines.length === 0) {
            container.innerHTML = '<span style="opacity: 0.6;">Keine Schlagzeilen verfügbar.</span>';
            return;
        }
        var html = '';
        for (var i = 0; i < headlines.length; i++) {
            var h = headlines[i];
            var score = typeof h.sentiment_score === 'number' ? h.sentiment_score.toFixed(2) : '--';
            var color = typeof h.sentiment_score === 'number' ? scoreColor(h.sentiment_score) : '#888';
            var feed = h.feed_name || h.source_name || '';
            html += '<div style="display:flex;align-items:baseline;gap:8px;padding:6px 0;border-bottom:1px solid rgba(128,128,128,0.15);">';
            html += '<span style="font-weight:700;min-width:42px;color:' + color + ';font-size:0.85em;">' + score + '</span>';
            html += '<span style="flex:1;font-size:0.9em;">' + h.headline + '</span>';
            if (feed) {
                html += '<span style="font-size:0.75em;opacity:0.55;white-space:nowrap;margin-left:6px;">' + feed + '</span>';
            }
            html += '</div>';
        }
        container.innerHTML = html;
    }

    function loadHeadlines() {
        fetch(HEADLINES_URL)
            .then(function(r) {
                if (!r.ok) throw new Error('HTTP ' + r.status);
                return r.json();
            })
            .then(function(data) {
                if (data.status === 'success') {
                    renderHeadlines(data.headlines);
                } else {
                    document.getElementById('headlines-list').innerHTML =
                        '<span style="opacity:0.6;">Schlagzeilen konnten nicht geladen werden.</span>';
                }
            })
            .catch(function() {
                document.getElementById('headlines-list').innerHTML =
                    '<span style="opacity:0.6;">Backend nicht erreichbar — Schlagzeilen nicht verfügbar.</span>';
            });
    }

    document.addEventListener('DOMContentLoaded', loadHeadlines);
})();
</script>
```

Hinweis zur URL: Der ESP32 läuft im Heimnetz, ruft aber das Backend direkt über seine externe URL ab — identisch zu docs/index.html. Keine apiUrl-Variable verfügbar in mood.js-Scope.
