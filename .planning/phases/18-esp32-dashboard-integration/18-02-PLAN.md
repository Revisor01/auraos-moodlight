**Schritt 2 — JavaScript ergänzen** in `loadOverview()` (nach dem bestehenden Block der scoreEl-Werte-Setzung, also nach `document.getElementById('barNeedle').style.left = scoreToPercent(score);` aber VOR der Formelzeile):

```javascript
      // VIS-01: Skalierungs-Kontext befüllen
      if (current.percentile !== undefined && current.thresholds && current.historical) {
        var pct      = Math.round(current.percentile * 100);
        var hist     = current.historical;
        var thresh   = current.thresholds;
        var histMin  = parseFloat(hist.min);
        var histMax  = parseFloat(hist.max);
        var histMed  = parseFloat(hist.median);
        var histSpan = histMax - histMin;

        // Perzentil-Badge
        document.getElementById('percentileVal').textContent = pct + '. Pzt.';
        document.getElementById('percentileDesc').textContent =
          'Der aktuelle Score liegt über ' + pct + ' % der Werte der letzten 7 Tage.';

        // Historische Min/Median/Max-Werte
        document.getElementById('histMin').textContent    = histMin.toFixed(3);
        document.getElementById('histMedian').textContent = histMed.toFixed(3);
        document.getElementById('histMax').textContent    = histMax.toFixed(3);
        document.getElementById('histCount').textContent  =
          (hist.count || '—') + ' Datenpunkte in den letzten 7 Tagen';

        // Farbverlauf-Balken: linke Kante = histMin, rechte Kante = histMax
        // Gesamtbreite des Tracks = [-1, +1] = 100 %
        if (histSpan > 0.001) {
          var leftPct  = ((histMin + 1) / 2 * 100).toFixed(1) + '%';
          var widthPct = (histSpan / 2 * 100).toFixed(1) + '%';
          var rangeEl  = document.getElementById('histBarRange');
          rangeEl.style.left  = leftPct;
          rangeEl.style.width = widthPct;

          // Score-Nadel innerhalb des Tracks positionieren
          document.getElementById('histNeedle').style.left = scoreToPercent(score);

          // Schwellwert-Ticks P20/P40/P60/P80
          var wrap = document.getElementById('histBarWrap');
          ['p20','p40','p60','p80'].forEach(function(key) {
            if (thresh[key] === undefined) return;
            var tPos = ((thresh[key] + 1) / 2 * 100).toFixed(1) + '%';
            var tick = document.createElement('div');
            tick.className = 'threshold-tick';
            tick.style.left = tPos;
            var lbl = document.createElement('div');
            lbl.className = 'threshold-label';
            lbl.style.left = tPos;
            lbl.textContent = key.toUpperCase() + ' ' + parseFloat(thresh[key]).toFixed(2);
            wrap.appendChild(tick);
            wrap.appendChild(lbl);
          });
        }

        // Fallback-Hinweis
        document.getElementById('fallbackHint').style.display =
          thresh.fallback ? 'block' : 'none';
      }
```
