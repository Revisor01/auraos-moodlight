Am Ende von docs/index.html, NACH dem letzten `</script>`-Tag und VOR `</body>` folgendes Script einfügen:

```html
    <script>
    // ===== FEED-VERGLEICH (PAGE-01) =====
    (function() {
        var API_BASE = 'https://analyse.godsapp.de/api/moodlight/feeds/trends';
        var ftCurrentDays = 7;

        function ftScoreColor(score) {
            if (score === null || score === undefined) return '#ced4da';
            if (score >= 0.30)  return '#28a745';
            if (score >= 0.10)  return '#6c3dbf';
            if (score >= -0.10) return '#17a2b8';
            if (score >= -0.30) return '#e67e22';
            return '#e74c3c';
        }

        function ftScoreClass(score) {
            if (score === null || score === undefined) return 'neu';
            if (score > 0.10)  return 'pos';
            if (score < -0.10) return 'neg';
            return 'neu';
        }

        function ftBarWidth(score) {
            if (score === null || score === undefined) return '0%';
            return ((score + 1) / 2 * 100).toFixed(1) + '%';
        }

        function ftRender(feeds) {
            var body = document.getElementById('ftRankBody');
            var countEl = document.getElementById('ftCount');
            if (!body) return;

            if (!feeds || feeds.length === 0) {
                body.innerHTML = '<p class="ft-loading">Keine Daten im gewählten Zeitfenster verfügbar.</p>';
                if (countEl) countEl.style.display = 'none';
                return;
            }

            if (countEl) {
                countEl.textContent = feeds.length + ' Feeds';
                countEl.style.display = 'inline-block';
            }

            var html = '<div class="feed-rank-grid">';
            for (var i = 0; i < feeds.length; i++) {
                var f = feeds[i];
                var score = (f.avg_score !== null && f.avg_score !== undefined) ? parseFloat(f.avg_score) : null;
                var scoreStr = score !== null ? (score >= 0 ? '+' : '') + score.toFixed(3) : '—';
                var barColor = ftScoreColor(score);
                var barW = ftBarWidth(score);
                var cls = ftScoreClass(score);
                var name = (f.feed_name || '—').replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;');

                html += '<div class="feed-rank-row">';
                html += '<span class="ft-rank-nr">' + (i + 1) + '</span>';
                html += '<span class="ft-rank-name">' + name + '</span>';
                html += '<div class="ft-bar-track"><div class="ft-bar-fill" style="width:' + barW + ';background:' + barColor + '"></div></div>';
                html += '<span class="ft-score ' + cls + '">' + scoreStr + '</span>';
                html += '</div>';
            }
            html += '</div>';
            body.innerHTML = html;
        }

        function ftLoad(days) {
            var body = document.getElementById('ftRankBody');
            if (body) body.innerHTML = '<p class="ft-loading">Lade Feed-Ranking…</p>';

            fetch(API_BASE + '?days=' + days)
                .then(function(r) {
                    if (!r.ok) throw new Error('HTTP ' + r.status);
                    return r.json();
                })
                .then(function(data) {
                    if (data.status === 'success' && Array.isArray(data.feeds)) {
                        ftRender(data.feeds);
                    } else {
                        if (body) body.innerHTML = '<p class="ft-error">Daten konnten nicht verarbeitet werden.</p>';
                    }
                })
                .catch(function() {
                    if (body) body.innerHTML = '<p class="ft-error">Backend nicht erreichbar — Feed-Ranking nicht verfügbar.</p>';
                });
        }

        window.ftSwitchWindow = function(days) {
            ftCurrentDays = days;
            document.querySelectorAll('.ft-window-btn').forEach(function(btn) {
                btn.classList.toggle('active', parseInt(btn.getAttribute('data-days')) === days);
            });
            ftLoad(days);
        };

        // Automatisch laden wenn Sektion sichtbar wird (IntersectionObserver)
        var section = document.getElementById('feed-trends');
        if (section && 'IntersectionObserver' in window) {
            var loaded = false;
            var observer = new IntersectionObserver(function(entries) {
                if (entries[0].isIntersecting && !loaded) {
                    loaded = true;
                    ftLoad(ftCurrentDays);
                    observer.disconnect();
                }
            }, { threshold: 0.1 });
            observer.observe(section);
        } else if (section) {
            // Fallback: sofort laden
            if (document.readyState === 'loading') {
                document.addEventListener('DOMContentLoaded', function() { ftLoad(ftCurrentDays); });
            } else {
                ftLoad(ftCurrentDays);
            }
        }
    })();
    </script>
```
