// ===== TAB-NAVIGATION =====
function showTab(tabId) {
  document.querySelectorAll('.tab-pane').forEach(p => p.classList.remove('active'));
  document.querySelectorAll('.tab-btn').forEach(b => b.classList.remove('active'));
  document.getElementById('tab-' + tabId).classList.add('active');
  document.querySelector('[data-tab="' + tabId + '"]').classList.add('active');

  if (tabId === 'overview' && !overviewLoaded) loadOverview();
  if (tabId === 'headlines' && !headlinesLoaded) loadHeadlines();
  if (tabId === 'feeds' && !feedsLoaded) loadFeeds();
}

document.querySelectorAll('.tab-btn').forEach(btn => {
  btn.addEventListener('click', () => showTab(btn.dataset.tab));

// ===== HILFSFUNKTIONEN =====
function escapeHtml(str) {
  const d = document.createElement('div');
  d.appendChild(document.createTextNode(str || ''));
  return d.innerHTML;
}

function formatDate(iso) {
  if (!iso) return '—';
  const d = new Date(iso);
  return d.toLocaleString('de-DE', {
    day: '2-digit', month: '2-digit', year: 'numeric',
    hour: '2-digit', minute: '2-digit'
  });
}

function relativeTime(iso) {
  if (!iso) return '—';
  const diff = Math.floor((Date.now() - new Date(iso)) / 1000);
  if (diff < 60)       return 'gerade eben';
  if (diff < 3600)     return 'vor ' + Math.floor(diff / 60) + ' Min.';
  if (diff < 86400)    return 'vor ' + Math.floor(diff / 3600) + ' Std.';
  return 'vor ' + Math.floor(diff / 86400) + ' Tagen';
}

function scoreClass(score) {
  if (score >= 0.30) return 'score-sehr-positiv';
  if (score >= 0.10) return 'score-positiv';
  if (score >= -0.20) return 'score-neutral';
  if (score >= -0.50) return 'score-negativ';
  return 'score-sehr-negativ';
}

function scoreToPercent(score) {
  // score ist [-1, +1], Balken geht von 0% bis 100%
  return ((score + 1) / 2 * 100).toFixed(1) + '%';
}

// ===== ÜBERSICHT =====
var overviewLoaded = false;

async function loadOverview() {
  try {
    const [currentRes, statsRes, feedsRes] = await Promise.all([
      fetch('/api/moodlight/current'),
      fetch('/api/moodlight/stats'),
      fetch('/api/moodlight/feeds')
    ]);
    const current = await currentRes.json();
    const stats   = await statsRes.json();
    const feeds   = await feedsRes.json();

    if (current.status === 'success') {
      const score = current.sentiment;
      document.getElementById('scoreValue').textContent = score.toFixed(2);
      document.getElementById('scoreValue').className = 'value ' + scoreClass(score);
      document.getElementById('scoreCategory').textContent = current.category || '';
      document.getElementById('lastUpdate').textContent = relativeTime(current.timestamp);
      document.getElementById('nextUpdate').textContent = 'Nächste Aktualisierung in ~30 Min.';
      document.getElementById('headerStatus').textContent =
        'Score: ' + score.toFixed(2) + ' · ' + (current.category || '') +
        ' · ' + relativeTime(current.timestamp);

      // Score-Balken positionieren
      document.getElementById('barNeedle').style.left = scoreToPercent(score);

      // Formelzeile — Roh-Durchschnitt rückrechnen (näherungsweise: arctanh(score) / 2)
      // Da wir den Roh-Wert nicht direkt haben, zeigen wir den Endwert und die Formel
      const fFinal = score.toFixed(3);
      // Näherung für Anzeige: atanh(score) / 2 = raw_average (wenn score = tanh(raw*2))
      const rawApprox = (Math.atanh(Math.max(-0.9999, Math.min(0.9999, score))) / 2).toFixed(3);
      const stretchedApprox = (parseFloat(rawApprox) * 2).toFixed(3);
      document.getElementById('fRaw').textContent = rawApprox;
      document.getElementById('fStretched').textContent = stretchedApprox;
      document.getElementById('fFinal').textContent = fFinal;
    }

    if (stats.status === 'success') {
      const avg24 = parseFloat(stats.avg_sentiment_24h || 0);
      document.getElementById('avg24h').textContent = avg24.toFixed(2);
      document.getElementById('avg24h').className = 'value ' + scoreClass(avg24);
    }

    if (feeds.status === 'success') {
      document.getElementById('feedCount').textContent = feeds.count;
    }

    overviewLoaded = true;
  } catch (e) {
    console.error('Fehler beim Laden der Übersicht:', e);
    document.getElementById('headerStatus').textContent = 'Fehler beim Laden';
  }
}

          textHtml +
          '<div class="headline-meta">' +

            escapeHtml(h.source_name || '') + ' · ' + relativeTime(h.analyzed_at) +
