    ```javascript
    // ===== MANUELLER ANALYSE-TRIGGER (CTRL-01/02/03) =====
    async function triggerAnalysis() {
      var btn       = document.getElementById('analyzeBtn');
      var resultEl  = document.getElementById('triggerResult');
      var errorEl   = document.getElementById('triggerError');

      // UI: Button deaktivieren, Spinner anzeigen
      btn.disabled = true;
      btn.innerHTML = '<span class="spinner"></span> Analysiere…';
      resultEl.style.display = 'none';
      errorEl.style.display  = 'none';

      try {
        var res  = await fetch('/api/moodlight/analyze/trigger', { method: 'POST' });
        var data = await res.json();

        if (res.ok && data.status === 'success') {
          // Erfolg: Meldung zeigen, Dashboard-Übersicht neu laden
          resultEl.textContent = 'Analyse abgeschlossen: Score ' +
            parseFloat(data.sentiment_score).toFixed(3) +
            ' (' + (data.category || '') + ')' +
            ', ' + (data.headlines_analyzed || '?') + ' Schlagzeilen' +
            ' in ' + (data.duration_seconds || '?') + 's';
          resultEl.style.display = 'block';

          // Übersicht neu laden (CTRL-03)
          overviewLoaded = false;
          await loadOverview();
        } else {
          // API-Fehler mit Meldung
          errorEl.textContent = 'Fehler: ' + (data.message || 'Unbekannter Fehler (HTTP ' + res.status + ')');
          errorEl.style.display = 'block';
        }
      } catch (e) {
        errorEl.textContent = 'Netzwerkfehler — API nicht erreichbar.';
        errorEl.style.display = 'block';
        console.error('Fehler beim Analyse-Trigger:', e);
      } finally {
        // UI: Button wieder aktivieren
        btn.disabled = false;
        btn.innerHTML = 'Jetzt analysieren';
      }
    }
    ```

    WICHTIG: Das `triggerAnalysis()` im `onclick`-Attribut des Buttons ist korrekt — keine Event-Listener-Probleme, da `triggerAnalysis` global im Script-Scope definiert ist (keine Module).
